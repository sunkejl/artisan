21    function sumElements ($array1, $array2)
22    {   
23        $ret = $array1;
24        
25        foreach ($array2 as $key => $element) {
26            if (isset ($ret[$key])) {
27                $ret[$key] += $element;
28            } else {
29                $ret[$key] = $element;
30            }
31        }
32        return $ret;
